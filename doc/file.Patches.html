def self.load(base, except: nil, only: nil)\
  patches = {
    string: ->(){ String.include Gemmy::Patches::StringPatch },
    symbol: ->(){ Symbol.include Gemmy::Patches::SymbolPatch },
    global: ->(){ Object.include Gemmy::Patches::GlobalPatch }
  }
  patches.each do |type, fn|
    next if except&.include? type
    next if only && !only.include?(type)
    base.class_exec { fn.call }
  end
  self
end
