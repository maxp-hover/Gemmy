Overview

The opposite of Hash#dig Takes a list of keys followed by a value to set

Example:

a = {a: {b: {}} }
a.bury(:a, :b, :c, 0)
puts a[:a][:b][:c]
=> 0

def self._bury(caller_hash, *args)
  if args.count < 2
    raise ArgumentError.new("2 or more arguments required")
  elsif args.count == 2
    caller_hash[args[0]] = args[1]
  else
    arg = args.shift
    caller_hash[arg] = {} unless caller_hash[arg]
    _bury(caller_hash[arg], *args) unless args.empty?
  end
  caller_hash
end
