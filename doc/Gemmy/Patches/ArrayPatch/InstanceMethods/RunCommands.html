def run_commands
  _eval_noun = Gemmy.patch("string/i/eval_noun")
                    .method(:_eval_noun)
  return self.flat_map do |cmds|
    cmds&.map do |cmd|
      eval(VerbLexicon.get cmd[:verb].to_sym).call(*(
        cmd[:nouns]&.map do |noun|
          _eval_noun.call(noun, self)
        end
      ).to_a.compact)
    end
  end.compact
end
