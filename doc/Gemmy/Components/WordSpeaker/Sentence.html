def speak(path: nil)
  pitch = DefaultPitch * diff_multiplier
  speed = DefaultSpeed * diff_factor
  gap = DefaultGap * diff_multiplier
  espeak_opts = "-v english-us -p #{pitch} -s #{speed}"
  if path
    espeak_opts += " -w #{path}"
  end
  `espeak #{espeak_opts} "#{sentence}"`
end
