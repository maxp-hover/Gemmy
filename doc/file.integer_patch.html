    module Factorial
      # facets
      def factorial
        return 1 if zero?
        f = 1
        2.upto(self) { |n| f *= n }
        f
      end
    end

    module MultipleOf
      # facets
      def multiple_of?(number)
        if number.zero?
          zero? ? true : false
        else
          self % number == 0
        end
      end
    end
